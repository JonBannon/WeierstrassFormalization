% Macros specific to the PDF (print) version of the blueprint.
%
% The web version's `\lean`, `\uses`, `\leanok`, etc. are provided by the
% `blueprint` plasTeX plugin; the print version compiles with a plain TeX
% engine, so these need dummy definitions here instead.

\newcommand{\lean}[1]{}
\newcommand{\discussion}[1]{}
\newcommand{\leanok}{}
\newcommand{\mathlibok}{}
\newcommand{\notready}{}
% Make sure that arguments of \uses and \proves are real labels, by using
% invisible refs: latex warns if the label is undefined, but nothing is
% shown in the pdf file.
\ExplSyntaxOn
\NewDocumentCommand{\uses}{m}
 {\clist_map_inline:nn{#1}{\vphantom{\ref{##1}}}%
  \ignorespaces}
\NewDocumentCommand{\proves}{m}
 {\clist_map_inline:nn{#1}{\vphantom{\ref{##1}}}%
  \ignorespaces}
\ExplSyntaxOff
